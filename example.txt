"""
Basic "what PyLazy might look like" example (proposed API).

Goals:
- Provider-agnostic connect()
- createFunction(prompt, signature=...) returns a normal Python callable
- Lazy generation (first call triggers LLM)
- Disk cache (so subsequent runs don't need the LLM)
"""

from pylazy import connect, createFunction


def main() -> None:
    # 1) Configure a provider.
    #
    # In practice you'd support:
    # - local: Ollama / llama.cpp / vLLM
    # - cloud: OpenAI / Groq / Gemini etc.
    #
    # For local testing (no paid key required), Ollama is a great default:
    connect(
        provider="ollama",
        base_url="http://localhost:11434",
        model="qwen2.5-coder:7b",  # example; user chooses what they have installed
    )

    # 2) Create a function from a description (+ optional signature).
    # This does NOT necessarily call the LLM yet; PyLazy can defer generation
    # until the function is first invoked.
    slugify = createFunction(
        "Convert a string into a URL-friendly slug. Lowercase it, replace spaces with '-', "
        "remove characters that aren't alphanumerics or '-'. Collapse repeated '-' and strip leading/trailing '-'.",
        signature="(text: str) -> str",
        cache=True,
        mode="lazy",  # lazy = generate on first call; eager could generate immediately
    )

    # 3) Use it like any other function.
    # First call triggers code generation + validation + compilation + caching.
    print(slugify("Hello, World! This is PyLazy."))

    # Subsequent calls should be fast and not hit the LLM (loaded from cache).
    print(slugify("  Spaces     and---punctuation!!!  "))

    # 4) Another function (shows reuse & caching).
    parse_csv = createFunction(
        "Parse a CSV string into a list of dicts using the first row as headers. "
        "Trim whitespace around headers and values. Ignore empty lines.",
        signature="(csv_text: str) -> list[dict[str, str]]",
        cache=True,
        mode="lazy",
    )

    csv_text = """name, age
    Alice, 30
    Bob,  29
    """

    print(parse_csv(csv_text))


if __name__ == "__main__":
    main()
